    /**
 * checking type of each object in the rules
 *
 * @param rules
 * @returns {boolean}
 *
 * function run(n, obj, str) {
 *   TypeChecker([
 *     'number',
 *     'player',
 *     String
 *  ], arguments); // true
 * }
 *
 * run(1, new Player(...), 'string');
 *
 */
export default function TypeChecker(rules, arg) {
  if (!(rules instanceof Array)) {
    throw new TypeError('rules must be an array');
  }

  if (!(arg instanceof Array)) {
    throw new TypeError('arg must be an array');
  }

  for (let i = 0, len = rules.length; i < len; ++i) {
    const t = arg[i], type = rules[i];
    if (Array === type) {
      if (t instanceof Array) continue;
    }
    if(typeof t == type);
  }

  return true;
};
